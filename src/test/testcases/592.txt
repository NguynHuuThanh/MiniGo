
    func binarySearch(arr [10] int, target int) int {
        var left = 0
        var right = 9
        for left <= right {
            var mid = (left + right) / 2
            if (arr[mid] == target) {
                return mid
            }
            if (arr[mid] < target) {
                left = mid + 1
            } else {
                right = mid - 1
            }
        }
        return -1
    }

        putInt(binarySearch(a, 19))
