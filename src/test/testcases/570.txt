
        func countOccurrences(arr [10] int, target int) int {
            var count = 0
            var i = 0
            for i < 10 {
                if (arr[i] == target) {
                    count += 1
                }
                i += 1
            }
            return count
        }
        
        func main() {
            var a [10] int = [10] int {1, 2, 3, 2, 4, 2, 5, 2, 6, 2}
            putInt(countOccurrences(a, 2))
        }
