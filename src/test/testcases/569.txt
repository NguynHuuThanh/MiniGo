
        func average(arr [5] int) float {
            var sum = 0
            var i = 0
            for i < 5 {
                sum += arr[i]
                i += 1
            }
            return float(sum) / 5.0
        }
        
        func main() {
            var a [5] int = [5] int {1, 2, 3, 4, 5}
            putFloat(average(a))
        }
